1. SQLite所支持的数据类型

在SQLite内部只支持 NULL、INTEGER、REAL(实数)、TEXT(文本)和BLOB（大二进制对象），除此之外SQLite还有一个特点，就是它允许把所有各种类型的数据保存到任何类型的字段中，所以我们不必关心声明该字段的时候所使用的数据类型。

2. 数据库操作语法

2.1 创建表

CREATE TABLE student(id INTEGER, name TEXT, score REAL);

CREATE TABLE student(id INTEGER, name TEXT, score REAL, PRIMARY KEY(id));

CREATE TABLE student(id INTEGER PRIMARY KEY, name TEXT, score REAL);

CREATE TABLE IF NOT EXISTS student(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT, score REAL);

CREATE TABLE IF NOT EXISTS student(id INTEGER PRIMARY KEY AUTOINCREMENT, name, score);

2.2 销毁表

DROP TABLE IF EXISTS t_sutdent;

DROP TABLE t_sutdent;

2.3 插入记录

INSERT INTO t_sutdent(score) VALUES(38);

INSERT INTO t_sutdent(score, name) VALUES(100, 'tt');

INSERT INTO t_sutdent(name, score) VALUES('zs', 99);

2.4 更新记录

UPDATE t_sutdent SET name = 'lnj';

2.5 删除记录

DELETE FROM t_sutdent;

2.6 条件语句

= /is / != / is not / > /< />= /<= 所支持的条件运算符
UPDATE t_sutdent SET name = 'xxx' WHERE score = 38 AND name ='ww';

UPDATE t_sutdent SET score = 99 WHERE name IS 'tt';

UPDATE t_sutdent SET score = 0 WHERE name = 'tt';

2.7 查询记录

SELECT * FROM t_sutdent WHERE score >= 60;

SELECT * FROM t_sutdent;

SELECT score FROM t_sutdent;

SELECT name, score FROM t_sutdent;

2.8 统计

SELECT count(*) FROM t_sutdent;

SELECT count(score) FROM t_sutdent WHERE score >= 60;

2.9 字段起别名

给字段起别名,隐藏内部真实存储的细节
给表起别名的用途:多表查询时, 如果多张表中都有相同名称的字段, 可以明确字段属于哪张表

SELECT stu.name , stu.score FROM t_sutdent AS stu;

SELECT name myName2, score myScore2 FROM t_sutdent;

SELECT name AS myName, score AS myScore FROM t_sutdent;

2.10 排序(默认升序)

SELECT * FROM t_sutdent ORDER BY score ASC;

先按照分数降序排序, 如果分数相同再按照id升序排序

SELECT * FROM t_sutdent ORDER BY score DESC, id DESC;

2.11 分页

SELECT * FROM t_sutdent LIMIT 10;

SELECT * FROM t_sutdent LIMIT 0, 10;

2.12 简单约束

CREATE TABLE IF NOT EXISTS t_student(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT, age INTEGER);

CREATE TABLE IF NOT EXISTS t_student(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, age INTEGER NOT NULL);

CREATE TABLE IF NOT EXISTS t_student(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT UNIQUE, age INTEGER);

CREATE TABLE IF NOT EXISTS t_student(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT, age INTEGER DEFAULT 1);

3. SQLite数据库常用函数

3.1 打开数据库

1.int sqlite3_open(
2.    const char *filename,   // 数据库的文件路径
3.    sqlite3 **ppDb          // 数据库实例
4.);

3.2 执行任何SQL语句

1.int sqlite3_exec(
2.    sqlite3*,                                  // 一个打开的数据库实例
3.    const char *sql,                           // 需要执行的SQL语句
4.    int (*callback)(void*,int,char**,char**),  // SQL语句执行完毕后的回调
5.    void *,                                    // 回调函数的第1个参数
6.    char **errmsg                              // 错误信息
7.);

3.3 检查SQL语句的合法性（查询前的准备）

1.int sqlite3_prepare_v2(
2.    sqlite3 *db,            // 数据库实例
3.    const char *zSql,       // 需要检查的SQL语句
4.    int nByte,              // SQL语句的最大字节长度
5.    sqlite3_stmt **ppStmt,  // sqlite3_stmt实例，用来获得数据库数据
6.    const char **pzTail
7.);

3.4 查询一行数据

1.int sqlite3_step(sqlite3_stmt*); // 如果查询到一行数据，就会返回SQLITE_ROW

3.5 利用stmt获得某一字段的值（字段的下标从0开始）

1.double sqlite3_column_double(sqlite3_stmt*, int iCol);  // 浮点数据
2.int sqlite3_column_int(sqlite3_stmt*, int iCol); // 整型数据
3.sqlite3_int64 sqlite3_column_int64(sqlite3_stmt*, int iCol); // 长整型数据
4.const void *sqlite3_column_blob(sqlite3_stmt*, int iCol); // 二进制文本数据
5.const unsigned char *sqlite3_column_text(sqlite3_stmt*, int iCol);  // 字符串数据

4.3.3 使用SQL语句执行查询

1.// 1.查询预编译
2.    /*
3.     第1个参数:需要查询的数据库
4.     第2个参数:需要执行的查询语句
5.     第3个参数:查询语句的长度, 如果传递一个小于0的数系统会自动计算查询语句的长度
6.     第4个参数:结果集对象
7.     */
8.    sqlite3_stmt *stmt;
9.    NSString *sql = @"SELECT * FROM student;";
10.    //执行查询语句,一般会调用sqlite3_prepare_v2() 预编译查询语句,然后循环调用sqlite3_step()函数取出查询结果集.
11.    //预编译SQL语句,将预编译结果保存到stmt中
12.    int queryResult = sqlite3_prepare_v2(self.db, sql.UTF8String, -1, &stmt, NULL);
13.    //判断是否预编译成功
14.    if (queryResult == SQLITE_OK) {
15.        NSLog(@"编译成功");
16.
17.        while (sqlite3_step(stmt) == SQLITE_ROW) {
18.            NSLog(@"有数据");
19.            const unsigned char *name = sqlite3_column_text(stmt, 1);
20.            double score = sqlite3_column_double(stmt, 2);
21.            NSLog(@"%s  %.1f", name, score);
22.        }
23.    }
24.    else{
25.        NSLog(@"没有预编译成功");
26.    }
